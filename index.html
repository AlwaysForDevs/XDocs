Player Movement with Joystick

Author: hazik

This script implements player movement using a joystick controller with smooth rotation towards the movement direction.

/* @Author hazik */ public SpatialObject player; // assign the player object in inspector public SUIJoystick joystick; // assign joystick in inspector public float moveSpeed = 10f; // movement speed public float turnSpeed = 5f; // rotation speed void start() { if (joystick == null || player == null) return; // --- Joystick input --- float x = joystick.getAxisX(); // horizontal (-1 left, 1 right) float y = joystick.getAxisY(); // vertical (-1 back, 1 forward) // --- Movement direction --- Vector3 forward = player.getTransform().getForward(); Vector3 right = player.getTransform().getRight(); Vector3 moveDir = forward.multiply(y).sum(right.multiply(x)); if (moveDir.length() > 0) { // normalize and scale by speed moveDir = moveDir.normalized().multiply(moveSpeed * Time.getDeltaTime()); // move player player.getTransform().translate(moveDir); // rotate smoothly toward movement direction Quaternion targetRot = Quaternion.lookRotation(moveDir, new Vector3(0,1,0)); player.getTransform().lerpRotation(targetRot, turnSpeed * Time.getDeltaTime()); } }
